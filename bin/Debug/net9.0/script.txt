function main():
    // Comment
    Term.print("Type anything: ")
    $var = Term.read()
    this.printer($var)
    
    if $var == "Hello World!":
        Term.print("Hello!")
    else if $var == "":
        Term.print("You type nothing...")
    else:
        Term.print("You typed: ", $var)
        
function printer($toprint):
    Term.print($toprint)
