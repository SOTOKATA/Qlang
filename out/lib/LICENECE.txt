; FILE WORK
finit:
    _lxc fpath;

    API fwork INCLUDE;

    fwork open fpath;

    assign file fwork create_pointer;

    ret_0:
        _0_lxc: 
            file;
        end;
    end;
end;

fopen:
    _lxc fpath;

    ret_0:
        _0_lxc: 
            CALL finit:
                _lxc fpath;
            end;
        end;
    end;
end;

fclose:
    _lxc xlcpointer;

    API fwork INCLUDE;

    fwork close_pointer xlcpointer;
end;
